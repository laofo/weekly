**json-schema-to-ts：从 JSON Schema 推断 TS 类型**  
<https://github.com/ThomasAribart/json-schema-to-ts>  

```js
const dogSchema = {
  type: 'object',
  properties: {
    name: { type: 'string' },
    age: { type: 'integer' },
    hobbies: { type: 'array', items: { type: 'string' } },
    favoriteFood: { enum: ['pizza', 'taco', 'fries'] },
  },
  required: ['name', 'age'],
};

type Dog = {
  name: string,
  age: number,
  hobbies?: string[],
  favoriteFood?: 'pizza' | 'taco' | 'fries',
};
```
