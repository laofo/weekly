```js
const colorSet: IColorSet = {
  base: {

    foreground: '#EFEFEF',
    color1: '#399EF4',
    color2: '#DA6771',
    color3: '#4EB071',
    color4: '#FFF099',
  },
};
generateTheme('My Theme', colorSet, path.join(__dirname, 'theme.json'));
```
